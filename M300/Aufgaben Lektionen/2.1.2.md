# 2.1.2 – Domänencontroller konfigurieren

## Legende
- LS = Linux Server

---

## 1. Service Script vorbereiten (LS)

nano /lib/systemd/system/samba-ad-dc.service

### Inhalt:
[Unit]
Description=Samba4 AD DC
After=network.target remote-fs.target nss-lookup.target

[Service]
Type=forking
ExecStart=/usr/sbin/samba -D --configfile=/etc/samba/smb.conf
PIDFile=/var/run/samba/samba.pid

[Install]
WantedBy=multi-user.target

---

## 2. Systemd neu laden (LS)
systemctl daemon-reload

---

## 3. Status prüfen (LS)
systemctl status samba-ad-dc

---

## 4. Maskierung entfernen (LS)
systemctl unmask samba-ad-dc

systemctl list-unit-files | grep samba

---

## 5. Dienst aktivieren (LS)
systemctl enable samba-ad-dc

systemctl daemon-reload

---

## 6. Samba Konfiguration vorbereiten (LS)
mv /etc/samba/smb.conf /home/vmadmin/

---

## 7. Domain erstellen (LS)
samba-tool domain provision

### Eingaben:
Realm: <DEINNAME>.M300  
Domain: <DEINNAME>  
Server Role: dc  
DNS backend: SAMBA_INTERNAL  
DNS forwarder: 192.168.210.1  
Passwort: Riethuesli>12345  

---

## 8. smb.conf prüfen (LS)
nano /etc/samba/smb.conf

Wichtige Werte:
dns forwarder = 192.168.210.1  
netbios name = DC1  
realm = <DEINNAME>.M300  
server role = active directory domain controller  

---

## 9. Alte Dienste stoppen (LS)
systemctl stop smbd nmbd winbind
systemctl disable smbd nmbd winbind

---

## 10. Samba AD starten (LS)
systemctl start samba-ad-dc

systemctl status samba-ad-dc

---

## 11. Neustart (LS)
reboot

---

## 12. Prozesse prüfen (LS)
ps ax | grep samba

---

## 13. Dienste prüfen (LS)
service smbd status
service nmbd status
service winbind status

→ müssen INAKTIV sein

---

## 14. Ports prüfen (LS)
ss -tulpn

---

## 15. DNS testen (LS)
host dc1
host -t SRV _kerberos._tcp.<deinname>.m300
host -t SRV _ldap._tcp.<deinname>.m300

---

## 16. SMB testen (LS)
smbclient -L dc1

---

## 17. Kerberos konfigurieren (LS)
nano /etc/krb5.conf

[libdefaults]
default_realm = <DEINNAME>.M300
dns_lookup_realm = false
dns_lookup_kdc = true

---

## 18. Kerberos testen (LS)
kinit administrator
klist

---

## 19. SMB mit Login testen (LS)
smbclient -L dc1 -U administrator

---

## 20. LDAP testen (LS)
ldapsearch -H ldap://dc1 -U administrator -x yes

---

## 21. nslookup testen (LS)
nslookup

---

## 22. Benutzer erstellen (LS)
samba-tool user create vmuser
samba-tool user create vmadmin

samba-tool user list

---

## 23. Gruppen testen (LS)
samba-tool group list
samba-tool group listmembers administrators

---

## 24. Passwortregeln setzen (LS)
samba-tool domain passwordsettings show
samba-tool domain passwordsettings set --max-pwd-age=49

---

## 25. LDAP Benutzer prüfen (LS)
ldapsearch -H ldap://dc1 "cn=vmuser" -U administrator

---

## 26. LDAP DC prüfen (LS)
ldapsearch -H ldap://dc1 "cn=dc1" -x yes

---

## 27. nslookup erweitern (LS)
nslookup
set type=any
<deinname>.m300
exit
